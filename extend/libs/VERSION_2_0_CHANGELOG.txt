HELPERS:
  from plugins import helpers
  
  # Валідація
  if helpers.is_email("user@example.com"):
      print("✓ Email валідний")
  
  # Конвертування
  age = helpers.to_int("25")
  
  # Форматування
  slug = helpers.slugify("My Blog Post!")  # my-blog-post
  
  # Файли
  helpers.write_file("data.json", content)
  
  # Кеш
  cache = helpers.SimpleCache(ttl=300)
  cache.set("key", value)

DSL:
  from plugins.dsl_integration import create_dsl_system
  
  context, executor = create_dsl_system()
  
  script = """
  set(name, "Іван")
  print("Привіт", $name)
  
  if($age > 18, print("Дорослий"))
  
  for(i, [1,2,3], print($i))
  """
  
  executor.execute_script(script)

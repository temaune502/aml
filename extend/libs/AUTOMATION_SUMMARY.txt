QUICK START:

# Console Automation
from plugins.console_automation import run_cmd, run_powershell

result = run_powershell('Get-Process explorer')
print(result.stdout)

# PowerShell Executor  
from plugins.powershell_executor import ps_get_processes

processes = ps_get_processes()
for proc in processes[:5]:
    print(proc['Name'])

# System Monitor
from plugins.system_monitor import get_system_metrics, get_top_processes

metrics = get_system_metrics()
print(f"CPU: {metrics.cpu_percent}%")

top = get_top_processes(limit=5)
for proc in top:
    print(f"{proc.name}: {proc.memory_percent}%")

# Windows Automation
from plugins.windows_automation import WindowsAutomation

services = WindowsAutomation.get_services(name_filter='Windows')
for svc in services[:3]:
    print(f"{svc['Name']}: {svc['Status']}")


📋 ОСНОВНІ ФУНКЦІЇ
═══════════════════════════════════════════════════════════════════════════════

Console Automation:
  run_cmd(command)                           # Запустити команду
  run_powershell(command)                    # PowerShell команда
  run_python(code)                           # Python код
  ConsoleAutomation.parse_json_output()      # JSON парсинг
  ConsoleAutomation.parse_table()            # Парсинг таблиці
  ConsoleAutomation.extract_regex()          # Витягнення по regex

PowerShell Executor:
  ps_run(command)                            # Запустити
  ps_get_processes()                         # Отримати процеси
  ps_stop_process(name)                      # Зупинити процес
  PowerShellExecutor.pipeline(*commands)     # Pipeline команди
  PowerShellExecutor.get_files()             # Отримати файли
  PowerShellExecutor.registry_read()         # Прочитати реєстр

System Monitor:
  get_system_metrics()                       # Метрики системи
  get_top_processes(limit)                   # Топ процесів
  kill_process_by_name(name)                 # Завершити процес
  SystemMonitor.monitor_continuous()         # Безперервний моніторинг

Windows Automation:
  WindowsAutomation.get_services()           # Список сервісів
  WindowsAutomation.start_service()          # Запустити сервіс
  WindowsAutomation.stop_service()           # Зупинити сервіс
  WindowsAutomation.registry_read()          # Прочитати реєстр
  WindowsAutomation.create_task()            # Створити завдання
  WindowsAutomation.shutdown()               # Вимкнення ПК
  WindowsAutomation.restart()                # Перезавантаження
